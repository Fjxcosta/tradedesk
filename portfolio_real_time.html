⚡ Configurar Cloudflare Worker
✅ Gratuito — 100.000 requisições/dia no plano free da Cloudflare. Leva ~3 minutos para configurar.
Passo a passo:
1Acesse workers.cloudflare.com e crie uma conta gratuita
2Clique em Create a Worker
3Apague todo o código e cole o código abaixo
4Clique em Deploy e copie a URL gerada (ex: https://meu-worker.usuario.workers.dev)
5Cole a URL no campo Worker URL acima e salve
📋 Código do Worker: 
export default {
  async fetch(request) {
    const url = new URL(request.url);
    const cors = {
      "Access-Control-Allow-Origin": "*",
      "Access-Control-Allow-Methods": "GET, POST, OPTIONS",
      "Access-Control-Allow-Headers": "Content-Type, x-api-key, anthropic-version, anthropic-beta",
    };

    if (request.method === "OPTIONS") {
      return new Response(null, { status: 204, headers: cors });
    }

    // Health check
    if (request.method === "GET" && url.pathname === "/") {
      return new Response(JSON.stringify({ status: "ok", service: "TradeDesk AI Proxy" }), {
        headers: { "Content-Type": "application/json", ...cors },
      });
    }

    // Proxy brapi.dev — GET /brapi/:ticker?token=...&range=...
    if (request.method === "GET" && url.pathname.startsWith("/brapi/")) {
      const ticker = url.pathname.replace("/brapi/", "");
      const qs = url.searchParams.toString();
      const brapiUrl = "https://brapi.dev/api/quote/" + ticker + (qs ? "?" + qs : "");
      const resp = await fetch(brapiUrl, {
        headers: { "User-Agent": "TradeDesk/3.0" }
      });
      const data = await resp.text();
      return new Response(data, {
        status: resp.status,
        headers: { "Content-Type": "application/json", ...cors },
      });
    }

    // Proxy Anthropic — POST /
    if (request.method === "POST") {
      const body = await request.text();
      const resp = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "x-api-key": request.headers.get("x-api-key") || "",
          "anthropic-version": request.headers.get("anthropic-version") || "2023-06-01",
        },
        body,
      });
      const data = await resp.text();
      return new Response(data, {
        status: resp.status,
        headers: { "Content-Type": "application/json", ...cors },
      });
    }

    return new Response("Not found", { status: 404, headers: cors });
  },
};
